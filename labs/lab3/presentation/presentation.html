using DifferentialEquations, Plots;

#диф уравнение модели боевых действий между рег войсками
function reg_part(u, p, t)
    x, y = u
    a, b, c, h = p
    dx = -a*x - b*y + sin(t+7) + 1
    dy = -c*x - h*y + cos(t+4) + 1
    return [dx, dy]
end

using DifferentialEquations, Plots;

#диф уравнение модели боевых действий между рег войсками
function reg_part(u, p, t)
    x, y = u
    a, b, c, h = p
    dx = -a*x - b*y + sin(2*t)
    dy = -c*x*y- h*y + 2*cos(t)
    return [dx, dy]
end
